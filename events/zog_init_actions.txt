event = {
    id = zog_init_events.0
    hide_window = yes
    is_triggered_only = yes

    trigger = {}

    immediate = {
        every_planet = {
            limit = {
                not_colonized_planet = yes
            }
            every_tile = {
                limit = {
                    is_empty_tile = yes
                }
                set_blocker = random
            }
            if = {
                limit = {
                    count_tile = {
                        limit = {
                            has_blocker = no
                        }
                        count < 3
                    }
                }
                while = {
                    count = 3
                    random_tile = {
                        limit = {
                            has_blocker = yes
                        }
                        remove_blocker = yes
                    }
                }
            }
        }
    }
}
