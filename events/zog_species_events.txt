planet_event = {
    id = zog_species_events.5
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        exists = event_target:species_researcher_country
        exists = event_target:species_subject_country
        count_pops = {
            count > 0
            limit = {
                is_growing = no
                is_robot_pop = no
                is_same_species = event_target:species_subject_country
            }
        }
    }

    immediate = {
        enable_special_project = {
            name = "ZOG_PROJECT_SPECIES_RESEARCH_A"
            location = this
            owner = event_target:species_researcher_country
        }
        event_target:species_researcher_country = {
            country_event = { id = zog_species_events.6 days = 5 }
        }
    }
}

country_event = {
    id = zog_species_events.6
    title = zog_species_events.6.title
    desc = zog_species_events.6.desc
    picture = GFX_evt_species_research_we
    show_sound = event_laboratory_sound
    location = event_target:species_subject_planet
    is_triggered_only = yes
    hide_window = no

    trigger = {}

    option = {
        name = zog_species_events.6.a
        custom_tooltip = zog_species_events.6.a.tooltip
        ai_chance = {
            factor = 100
        }
    }
    option = {
        name = zog_species_events.6.b
        custom_tooltip = zog_species_events.6.b.tooltip
        trigger = {
            has_special_project = "ZOG_PROJECT_SPECIES_RESEARCH_A"
        }
        ai_chance = {
            factor = 25
        }
        hidden_effect = {
            abort_special_project = {
                type = "ZOG_PROJECT_SPECIES_RESEARCH_A"

    }
