building_windmill_01 = {
    is_listed = yes
    empire_unique = no
    planet_unique = no
    requires_pop = no
    event_building = no
    can_demolish = yes
    icon = "building_windmill_01"
    tags = {}

    base_buildtime = 550

    prerequisites = {
        "tech_sandsea_construction"
    }

    upgrades = {
        "building_windmill_02"
    }

    show_tech_unlock_if = {
        NOT = {
            has_authority = "auth_machine_intelligence"
        }
    }

    cost = {
        minerals = 250
        energy = 250
    }

    required_resources = {
        minerals = 1
    }

    produced_resources = {
        energy = 2
    }

    planet_modifier = {
        planet_colony_development_speed_mult = 0.01
    }

    potential = {
        owner = {
            NOT = {
                has_authority = "auth_machine_intelligence"
            }
        }
        planet = {
            OR = {
                is_planet_class = pc_desert
                is_planet_class = pc_arid
                is_planet_class = pc_savannah
                # @depend "Planetary Diversity" (id:819148835)
                is_planet_class = pc_steppe
                is_planet_class = pc_hadesert
                is_planet_class = pc_sandsea
            }
        }
    }

    destroy_if = {
        OR = {
            AND = {
                exists = owner
                owner = {
                    has_authority = "auth_machine_intelligence"
                }
            }
            planet = {
                NOR = {
                    is_planet_class = pc_desert
                    is_planet_class = pc_arid
                    is_planet_class = pc_savannah
                    # @depend "Planetary Diversity" (id:819148835)
                    is_planet_class = pc_steppe
                    is_planet_class = pc_hadesert
                    is_planet_class = pc_sandsea
                }
            }
        }
    }

    allow = {
        AND = {
            custom_tooltip = {
                fail_text = "requires_building_capital_1_or_hab"
                planet = {
                    OR = {
                        has_building = "building_hab_capital"
                        has_building = "building_capital_1"
                        has_building = "building_capital_2"
                        has_building = "building_capital_3"
                    }
                }
            }
        }
    }

    ai_allow = {
        is_slave_tile_or_planet = yes
        is_robot_tile_or_planet = no
    }

    ai_weight = {
        factor = 65
    }

    ai_replace = yes
}

    empire_unique = no

    tags = {}

    base_buildtime = 1250

    }

    upgrades = {
        "building_thermal_borehole_02"
    }

        NOT = {

    }

    cost = {
        minerals = 1250
        energy = 1700
    }

    required_resources = {
        energy = 3
    }

    produced_resources = {
        minerals = 7
    }

    produced_resource_trigger = {
        modifier = {
            AND = {
                has_technology = "tech_electric_theory"
                NOT = { has_technology = "tech_electric_science" }
            }
            resources = {
                energy = 3
            }
        }
        modifier = {
            AND = {
                has_technology = "tech_electric_science"
                NOT = { has_technology = "tech_electric_doctrine" }
            }
            resources = {
                energy = 4
            }
        }
        modifier = {
            AND = {
                has_technology = "tech_electric_doctrine"
                NOT = { has_technology = "tech_electric_transcendence" }
            }
            resources = {
                energy = 5
            }
        }
        modifier = {
            AND = {
                has_technology = "tech_electric_transcendence"
                NOT = { has_technology = "tech_electric_infinitum" }
            }
            resources = {
                energy = 6
            }
        }
        modifier = {
            has_technology = "tech_electric_infinitum"
            resources = {
                energy = 7
            }
        }
    }

    planet_modifier = {
        slave_mineral_output = 0.10
        slave_food_output = 0.05
        tile_resource_minerals_mult = 0.10
    }

    planet_modifier_with_pop_trigger = {
        key = "building_thermal_borehole_modifier"
        potential = {
            always = yes
        }
        modifier = {
            pop_happiness = -0.05
            pop_environment_tolerance = -0.10
            pop_robot_production_output = 0.05
        }
    }

    country_modifier = {
        max_minerals = 1000
    }

    potential = {
        owner = {
            NOT = {
                has_authority = "auth_machine_intelligence"
            }
        }
        planet = {
            OR = {
                is_planet_class = pc_desert
                is_planet_class = pc_arid
                # @depend "Planetary Diversity" (id:819148835)
                is_planet_class = pc_hadesert
                is_planet_class = pc_sandsea
                is_planet_class = pc_geothermal
            }
        }
    }

    destroy_if = {
        OR = {
            AND = {
                exists = owner
                owner = {
                    has_authority = "auth_machine_intelligence"
                }
            }
            planet = {
                NOR = {
                    is_planet_class = pc_desert
                    is_planet_class = pc_arid
                    # @depend "Planetary Diversity" (id:819148835)
                    is_planet_class = pc_hadesert
                    is_planet_class = pc_sandsea
                    is_planet_class = pc_geothermal
                }
            }
        }
    }

    allow = {
        AND = {
            custom_tooltip = {
                fail_text = "requires_resource_minerals_1"
                tile = {
                    has_resource = {
                        type = minerals
                        amount > 0
                    }
                }
            }
            custom_tooltip = {
                fail_text = "requires_enslaved_or_robotic_pop"
                tile = {
                    OR = {
                        has_enslaved_pop = yes
                        has_robotic_pop = yes
                    }
                }
            }
        }
    }

    ai_allow = {
        is_slave_tile_or_planet = yes
        is_robot_tile_or_planet = yes
    }

    ai_weight = {
        weight = 1

        modifier = {
            factor = 10
            tile = {
                has_resource = {
                    type = minerals
                    amount < 3
                }
            }
        }
        modifier = {
            factor = 25
            tile = {
                has_resource = {
                    type = minerals
                    amount = 3
                }
            }
        }
        modifier = {
            factor = 100
            tile = {
                has_resource = {
                    type = minerals
                    amount > 3
                }
            }
        }
        modifier = {
            factor = 0.1
            tile = {
                has_resource = {
                    type = energy
                    amount > 3
                }
            }
        }
    }

    ai_replace = yes
}
