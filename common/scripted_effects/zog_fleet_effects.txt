# This = Solar System
create_system_crystal_fleet_01 = {
    random_system_planet = {
        limit = { is_star = no }
        event_target:crystal_country = {
            create_fleet = {
                name = "NAME_Sapphire_Lurkers"
                effect = {
                    set_owner = event_target:crystal_country
                    while = {
                        count = 10
                        create_ship = {
                            name = ""
                            design = "NAME_Large_Crystal_Entity_Blue"
                        }
                    }
                    while = {
                        count = 30
                        create_ship = {
                            name = ""
                            design = "NAME_Small_Crystal_Entity_Blue"
                        }
                    }
                    set_formation_scale = 4.0
                    set_location = PREVPREV
                    set_fleet_stance = aggressive
                    set_aggro_range_measure_from = self
                    set_aggro_range = 150
                }
            }
        }
    }
}
